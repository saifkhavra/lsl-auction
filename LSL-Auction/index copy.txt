.then(response => response.text())

.then(data => {

  const rows =
  data
  .trim()
  .split("\n")

const cols = row.split(",").map(col =>
  col.replace(/^"|"$/g, "").trim()
);

  location: cols[2],
  role: cols[3],
  batting: cols[4],
  bowling: cols[5],
  category: cols[6],
  basePrice: cols[7],
  img: cols[8]
};

  });

  renderPlayers(players);
});

/* FILTERS */

function filterPlayers(){

  const search =
  document
  .getElementById("searchInput")
  .value
  .toLowerCase();

  const role =
  document
  .getElementById("roleFilter")
  .value;

  const location =
  document
  .getElementById("locationFilter")
  .value;

  const filtered =
  players.filter(p => {

    return (

      p.name
      .toLowerCase()
      .includes(search)

      &&

      (
        role === ""
        ||
        p.role === role
      )

      &&

      (
        location === ""
        ||
        p.location === location
      )
    );
  });

  renderPlayers(filtered);
}

let idleTimer;

const idleOverlay =
document.getElementById("idleOverlay");

const idleVideo =
document.getElementById("idleVideo");

/* VIDEOS */

const idleVideos = [

  "videos/team1.mp4",
  "videos/team2.mp4"
];

let currentVideo = 0;

let videoLoop;

/* START IDLE */

function startIdleMode(){

  console.log("IDLE MODE STARTED");

  idleOverlay.style.display = "flex";

  setTimeout(() => {

    idleOverlay.style.opacity = "1";

  }, 50);

  playNextVideo();
}

/* STOP IDLE */

function stopIdleMode(){

  idleOverlay.style.display = "none";

  idleVideo.pause();

  idleVideo.currentTime = 0;

  clearTimeout(videoLoop);
}

/* PLAY VIDEOS */

function playNextVideo(){

console.log(idleVideos[currentVideo]);

  idleVideo.src =
  idleVideos[currentVideo];

  idleVideo.volume = 1;

  idleVideo.muted = true;

const playPromise =
idleVideo.play();

if(playPromise !== undefined){

  playPromise
  .then(() => {

    console.log("Video Playing");

  })

  .catch(error => {

    console.log("Autoplay blocked:", error);

    idleVideo.muted = true;

    idleVideo.play();
  });
}

  currentVideo =
  (currentVideo + 1)
  %
  idleVideos.length;

  idleVideo.onended = () => {

    videoLoop =
    setTimeout(() => {

      playNextVideo();

    }, 3000); // 3 sec gap
  };
}

/* RESET TIMER */

function resetIdleTimer(){

  clearTimeout(idleTimer);

  stopIdleMode();

  idleTimer =
  setTimeout(startIdleMode, 5000);
}

/* USER ACTIVITY */

[
  "mousemove",
  "mousedown",
  "keypress",
  "touchstart",
  "scroll"
]

.forEach(event => {

  document.addEventListener(
    event,
    resetIdleTimer
  );
});

/* START TIMER */

resetIdleTimer();
