<script>

let players = [];
let selectedPlayer = null;

const teams = [
{name:"Langha Tigers",purse:100000,used:0,players:[]},
{name:"Langha Lions",purse:100000,used:0,players:[]},
{name:"Langha Bulls",purse:100000,used:0,players:[]},
{name:"Langha Eagles",purse:100000,used:0,players:[]},
{name:"Langha Panthers",purse:100000,used:0,players:[]},
{name:"Langha Unicorns",purse:100000,used:0,players:[]},
{name:"Langha Sharks",purse:100000,used:0,players:[]},
{name:"Langha Wolves",purse:100000,used:0,players:[]}
];

function showPage(pageId){
["playersPage","dashboardPage","teamsPage"].forEach(id=>{
document.getElementById(id).style.display="none";
});
document.getElementById(pageId).style.display="block";
}

fetch("players.csv")
.then(response => response.text())
.then(data => {

const rows = data.trim().split("\n").slice(1);

players = rows.map(row => {
const cols = row.split(",").map(col => col.replace(/^"|"$/g, "").trim());

return {
name: cols[0],
age: cols[1],
location: cols[2],
role: cols[3],
batting: cols[4],
bowling: cols[5],
category: cols[6],
basePrice: cols[7],
img: cols[8]
};
});

renderPlayers(players);
loadDashboard();
loadTeamsPage();
});

function renderPlayers(list){

const grid = document.getElementById("playerGrid");
grid.innerHTML = "";

list.forEach(p => {

const card = document.createElement("div");
card.className = "card";

card.innerHTML = `
<img src="${p.img}" onerror="this.src='https://via.placeholder.com/250'">
<div class="card-content">
<h3>${p.name}</h3>
<div class="badge">${p.role}</div>
<div class="badge">📍 ${p.location}</div>
<div class="badge price-badge">💰 ₹${p.basePrice}</div>
${p.sold ? `<div class="badge" style="background:limegreen;color:white;">SOLD → ${p.soldTo}</div>` : ""}
</div>`;

card.onclick = () => openPlayerModal(p);

grid.appendChild(card);
});
}

function openPlayerModal(p){

selectedPlayer = p;

const modal = document.getElementById("modal");
const modalContent = document.getElementById("modalContent");

modal.style.display = "flex";

modalContent.innerHTML = `
<div class="modal-left">
<img src="${p.img}" onerror="this.src='https://via.placeholder.com/400'">
</div>

<div class="modal-right">
<h2>${p.name}</h2>
<div class="player-detail"><b>Age:</b> ${p.age}</div>
<div class="player-detail"><b>Location:</b> ${p.location}</div>
<div class="player-detail"><b>Role:</b> ${p.role}</div>
<div class="player-detail"><b>Batting:</b> ${p.batting}</div>
<div class="player-detail"><b>Bowling:</b> ${p.bowling}</div>

<div class="player-detail">
<select id="modalTeamSelect" style="width:100%;padding:12px;border:none;border-radius:12px;">
<option value="">Select Team</option>
${teams.map(team => `<option value="${team.name}">${team.name}</option>`).join("")}
</select>
</div>

<div class="player-detail">
<input id="modalBidAmount" type="number" value="${p.basePrice}" style="width:100%;padding:12px;border:none;border-radius:12px;">
</div>

<button class="close-btn" onclick="sellPlayerFromModal()" style="background:linear-gradient(90deg,gold,orange);color:black;">🔥 SOLD</button>

<button class="close-btn" onclick="unsoldPlayer()" style="background:linear-gradient(90deg,#555,#222);color:white;">UNSOLD</button>

<button class="close-btn" onclick="closeModal()" style="background:linear-gradient(90deg,#444,#111);color:white;">❌ CLOSE</button>

</div>`;
}

function closeModal(){
document.getElementById("modal").style.display = "none";
selectedPlayer = null;
}

window.onclick = function(event){
if(event.target && event.target.id === "modal"){
closeModal();
}
}

function sellPlayerFromModal(){

if(!selectedPlayer) return;

const selectedTeamName = document.getElementById("modalTeamSelect").value;
const soldAmount = parseInt(document.getElementById("modalBidAmount").value);

if(!selectedTeamName){
alert("Select Team");
return;
}

const team = teams.find(t => t.name === selectedTeamName);

if(team.purse < soldAmount){
alert("Not enough purse");
return;
}

selectedPlayer.sold = true;
selectedPlayer.soldTo = team.name;
selectedPlayer.soldPrice = soldAmount;

team.players.push(selectedPlayer);
team.used += soldAmount;
team.purse -= soldAmount;

renderPlayers(players);
loadDashboard();
loadTeamsPage();

const playerName = selectedPlayer.name;

closeModal();

alert(`${playerName} sold to ${team.name} for ₹${soldAmount}`);
}

function unsoldPlayer(){

if(!selectedPlayer) return;

selectedPlayer.sold = false;
selectedPlayer.soldTo = "";
selectedPlayer.soldPrice = "";

renderPlayers(players);
loadDashboard();
loadTeamsPage();

const playerName = selectedPlayer.name;

closeModal();

alert(`${playerName} remains UNSOLD`);
}

function loadDashboard(){

const dashboard = document.getElementById("teamsDashboard");
dashboard.innerHTML = "";

teams.forEach(team => {

dashboard.innerHTML += `
<div class="team-box">
<h2>${team.name}</h2>
<div class="team-stat">💰 Remaining Purse: ₹${team.purse}</div>
<div class="team-stat">💸 Used Purse: ₹${team.used}</div>
<div class="team-stat">👥 Players: ${team.players.length}</div>
</div>`;
});
}

function loadTeamsPage(){

const squads = document.getElementById("teamsSquads");
squads.innerHTML = "";

teams.forEach(team => {

squads.innerHTML += `
<div class="team-box">
<h2>${team.name}</h2>
<div class="team-stat">
${team.players.length
? team.players.map(player => `🏏 ${player.name} - ₹${player.soldPrice}<br>`).join("")
: "No Players Yet"}
</div>
</div>`;
});
}

let idleTimer;
const idleOverlay = document.getElementById("idleOverlay");
const idleVideo = document.getElementById("idleVideo");
const idleVideos = ["videos/team1.mp4","videos/team2.mp4"];
let currentVideo = 0;

function startIdleMode(){
idleOverlay.style.display = "flex";
setTimeout(()=>{idleOverlay.style.opacity="1";},50);
playNextVideo();
}

function stopIdleMode(){
idleOverlay.style.display = "none";
idleVideo.pause();
idleVideo.currentTime = 0;
}

function playNextVideo(){
idleVideo.src = idleVideos[currentVideo];
idleVideo.muted = true;
idleVideo.play();
currentVideo = (currentVideo + 1) % idleVideos.length;
idleVideo.onended = ()=>{
setTimeout(()=>{playNextVideo();},3000);
};
}

function resetIdleTimer(){
clearTimeout(idleTimer);
stopIdleMode();
idleTimer = setTimeout(startIdleMode,5000);
}

["mousemove","mousedown","keypress","touchstart","scroll"]
.forEach(event => {
document.addEventListener(event,resetIdleTimer);
});

resetIdleTimer();
