function sellPlayerFromModal(){

if(!selectedPlayer) return;

const selectedTeamName =
document.getElementById("modalTeamSelect").value;

const soldAmount =
parseInt(
document.getElementById("modalBidAmount").value
);

if(!selectedTeamName){

document.getElementById("hammerSound").play();

alert("Select Team");

document.getElementById("hammerSound").play();

createConfetti();

alert(
`${playerName} sold to ${team.name} for ₹${soldAmount}`
);

return;
}

const team =
teams.find(t => t.name === selectedTeamName);

if(!team){

return;
}

/* =========================
   REGIONAL LIMIT CHECK
========================= */

const sameRegionPlayers =
team.players.filter(player =>

player.location.trim().toLowerCase()
===

selectedPlayer.location.trim().toLowerCase()

).length;

/* MAX 5 PLAYERS */

if(sameRegionPlayers >= 5){

alert(
`❌ ${team.name} already has maximum 5 players from ${selectedPlayer.location}`
);

return;
}

/* =========================
   PURSE CHECK
========================= */

if(team.purse < soldAmount){

alert("Not enough purse");

return;
}

/* =========================
   SOLD UPDATE
========================= */

selectedPlayer.sold = true;

selectedPlayer.soldTo = team.name;

selectedPlayer.soldPrice = soldAmount;

/* ADD PLAYER TO TEAM */

team.players.push(selectedPlayer);

team.used += soldAmount;

team.purse -= soldAmount;

/* =========================
   REFRESH UI
========================= */

renderPlayers(players);

loadDashboard();

loadTeamsPage();

const playerName = selectedPlayer.name;

closeModal();

/* SUCCESS MESSAGE */

alert(
`${playerName} sold to ${team.name} for ₹${soldAmount}`
);
}

alert(`${playerName} remains UNSOLD`);
}

let idleTimer;
const idleOverlay = document.getElementById("idleOverlay");
const idleVideo = document.getElementById("idleVideo");
const idleVideos = ["videos/team1.mp4","videos/team2.mp4"];
let currentVideo = 0;

function startIdleMode(){
idleOverlay.style.display = "flex";
setTimeout(()=>{idleOverlay.style.opacity="1";},50);
playNextVideo();
}

function stopIdleMode(){
idleOverlay.style.display = "none";
idleVideo.pause();
idleVideo.currentTime = 0;
}

function playNextVideo(){
idleVideo.src = idleVideos[currentVideo];
idleVideo.muted = true;
idleVideo.play();
currentVideo = (currentVideo + 1) % idleVideos.length;
idleVideo.onended = ()=>{
setTimeout(()=>{playNextVideo();},3000);
};
}

function resetIdleTimer(){
clearTimeout(idleTimer);
stopIdleMode();
idleTimer = setTimeout(startIdleMode,5000);
}

["mousemove","mousedown","keypress","touchstart","scroll"]
.forEach(event => {
document.addEventListener(event,resetIdleTimer);
});

function createConfetti(){

for(let i = 0; i < 180; i++){

const confetti =
document.createElement("div");

/* CREATE CONFETTI */

confetti.style.position = "fixed";

confetti.style.width =
Math.random() * 12 + 6 + "px";

confetti.style.height =
Math.random() * 12 + 6 + "px";

/* COLORS */

const colors = [

"#FFD700", // gold
"#00E5FF", // cyan
"#FF00CC", // pink
"#00FF88", // green
"#FFFFFF"  // white

];

confetti.style.background =
colors[
Math.floor(
Math.random() * colors.length
)
];

/* RANDOM POSITION */

confetti.style.left =
Math.random() * 100 + "vw";

confetti.style.top = "-20px";

/* STYLE */

confetti.style.borderRadius =
Math.random() > 0.5
? "50%"
: "2px";

confetti.style.zIndex = "999999";

confetti.style.pointerEvents = "none";

/* GLOW */

confetti.style.boxShadow =
"0 0 12px rgba(255,255,255,0.6)";

/* ADD TO BODY */

document.body.appendChild(confetti);

/* RANDOM ANIMATION */

const xMove =
(Math.random() - 0.5) * 500;

const rotate =
Math.random() * 1080;

const duration =
Math.random() * 3 + 3;

/* ANIMATE */

confetti.animate([

{
transform:
"translateY(0px) translateX(0px) rotate(0deg)",
opacity:1
},

{
transform:
`translateY(110vh) translateX(${xMove}px) rotate(${rotate}deg)`,

opacity:0
}

],{

duration: duration * 1000,

easing:"cubic-bezier(.21,.98,.6,.99)"
});

/* REMOVE */

setTimeout(() => {

confetti.remove();

}, duration * 1000);

}
}

function createConfetti(){

/* FLASH EFFECT */

const flash =
document.createElement("div");

flash.style.position = "fixed";
flash.style.inset = "0";
flash.style.background =
"rgba(255,255,255,0.15)";
flash.style.zIndex = "999998";
flash.style.pointerEvents = "none";
flash.style.opacity = "1";
flash.style.transition =
"opacity 0.8s ease";

document.body.appendChild(flash);

setTimeout(()=>{
flash.style.opacity = "0";
},100);

setTimeout(()=>{
flash.remove();
},900);

/* GLITTER BURST */

for(let i = 0; i < 320; i++){

const piece =
document.createElement("div");

const size =
Math.random() * 14 + 6;

piece.style.width =
size + "px";

piece.style.height =
size + "px";

piece.style.position = "fixed";

piece.style.left =
Math.random() * 100 + "vw";

piece.style.top = "-20px";

const colors = [
"#FFD700",
"#FFF5B8",
"#00E5FF",
"#FF00CC",
"#00FF88",
"#FFFFFF",
"#FFAA00"
];

const color =
colors[
Math.floor(
Math.random() * colors.length
)
];

piece.style.background = color;

piece.style.borderRadius =
Math.random() > 0.5
? "50%"
: "3px";

piece.style.boxShadow = `
0 0 12px ${color},
0 0 24px ${color},
0 0 40px rgba(255,255,255,0.8)
`;

piece.style.zIndex = "999999";
piece.style.pointerEvents = "none";

document.body.appendChild(piece);

const xMove =
(Math.random() - 0.5) * 900;

const rotate =
Math.random() * 1440;

const duration =
Math.random() * 4 + 3;

piece.animate([

{
transform:
`
translateY(0px)
translateX(0px)
rotate(0deg)
scale(1)
`,
opacity:1
},

{
transform:
`
translateY(120vh)
translateX(${xMove}px)
rotate(${rotate}deg)
scale(0.4)
`,
opacity:0
}

],{

duration:
duration * 1000,

easing:
"cubic-bezier(.12,.89,.32,1)"
});

setTimeout(()=>{
piece.remove();
},duration * 1000);
}

/* CENTER EXPLOSION */

for(let i = 0; i < 80; i++){

const sparkle =
document.createElement("div");

sparkle.style.position = "fixed";

sparkle.style.left = "50%";
sparkle.style.top = "50%";

sparkle.style.width =
Math.random() * 10 + 4 + "px";

sparkle.style.height =
sparkle.style.width;

sparkle.style.borderRadius = "50%";

sparkle.style.background =
"#FFD700";

sparkle.style.boxShadow = `
0 0 18px gold,
0 0 40px white
`;

sparkle.style.zIndex = "999999";

document.body.appendChild(sparkle);

const angle =
Math.random() * 360;

const distance =
Math.random() * 600 + 150;

const x =
Math.cos(angle) * distance;

const y =
Math.sin(angle) * distance;

sparkle.animate([

{
transform:
"translate(0px,0px) scale(1)",
opacity:1
},

{
transform:
`translate(${x}px,${y}px) scale(0)`,
opacity:0
}

],{

duration:1800,
easing:"ease-out"
});

setTimeout(()=>{
sparkle.remove();
},1800);
}
}
