function showPage(pageId){

[
"playersPage",
"dashboardPage",
"teamsPage"
].forEach(id => {

document.getElementById(id).style.display = "none";

});

document.getElementById(pageId).style.display = "block";

}

fetch("players.csv")

.then(response => response.text())

.then(data => {

const rows =
data.trim().split("\n").slice(1);

players = rows.map(row => {

const cols =
row.split(",").map(col =>
col.replace(/^"|"$/g,"").trim()
);

return {

name: cols[0],
age: cols[1],
location: cols[2],
role: cols[3],
batting: cols[4],
bowling: cols[5],
category: cols[6],
img: cols[7],

sold:false,
soldTo:"",
soldPrice:0

};

});

loadSoldPlayers();

});

function loadSoldPlayers(){

teams.forEach(team => {

team.players = [];
team.used = 0;
team.purse = 100000;

});

fetch("sold_players.csv")

.then(response => response.text())

.then(data => {

const rows =
data.trim().split("\n").slice(1);

rows.forEach(row => {

const cols =
row.split(",").map(c =>
c.replace(/^"|"$/g,"").trim()
);

const playerName = cols[0];
const auctionStatus = cols[6];
const soldPrice = parseInt(cols[7]) || 0;
const teamName = cols[8];

if(
auctionStatus.toLowerCase() !== "sold"
){
return;
}

const player =
players.find(
p =>
p.name.toLowerCase().trim()
===
playerName.toLowerCase().trim()
);

const team =
teams.find(
t =>
t.name.toLowerCase().trim()
===
teamName.toLowerCase().trim()
);

if(player && team){

player.sold = true;
player.soldTo = team.name;
player.soldPrice = soldPrice;

team.players.push(player);

team.used += soldPrice;
team.purse -= soldPrice;

}

});

renderPlayers(players);
loadDashboard();
loadTeamsPage();

});

}

function renderPlayers(list){

const grid =
document.getElementById("playerGrid");

grid.innerHTML = "";

list.forEach(player => {

const card =
document.createElement("div");

card.className = "card";

card.innerHTML = `

<img
src="${player.img}"
onerror="this.src='https://via.placeholder.com/250x300?text=No+Image'"
>

<div class="card-content">

<h3>${player.name}</h3>

<div class="badge">
🏏 ${player.role}
</div>

<div class="badge">
📍 ${player.location}
</div>

<div class="badge">
🎂 ${player.age}
</div>

<div class="badge">
⭐ ${player.category}
</div>

${
player.sold

?

`

<div class="badge sold-badge">
🔥 SOLD → ${player.soldTo}
</div>

<div class="badge price-badge">
💰 ₹${player.soldPrice}
</div>

`

:

""

}

</div>

`;

grid.appendChild(card);

});

}

}

function filterPlayers(){

const search =
document.getElementById("searchInput")
.value
.toLowerCase();

const role =
document.getElementById("roleFilter")
.value;

const location =
document.getElementById("locationFilter")
.value;

const filtered =
players.filter(player => {

return (

player.name.toLowerCase().includes(search)

&&

(
role === ""
||
player.role === role
)

&&

(
location === ""
||
player.location === location
)

);

});

renderPlayers(filtered);

}

/* IDLE VIDEO */

let idleTimer;

const idleOverlay =
document.getElementById("idleOverlay");

const idleVideo =
document.getElementById("idleVideo");

const idleVideos = [

"videos/team1.mp4",
"videos/team2.mp4"

];

let currentVideo = 0;

function startIdleMode(){

idleOverlay.style.display = "flex";

playNextVideo();

}

function stopIdleMode(){

idleOverlay.style.display = "none";

idleVideo.pause();

idleVideo.currentTime = 0;

}

function playNextVideo(){

idleVideo.src =
idleVideos[currentVideo];

idleVideo.play();

currentVideo =
(currentVideo + 1)
%
idleVideos.length;

idleVideo.onended = () => {

setTimeout(() => {

playNextVideo();

},3000);

};

}

function resetIdleTimer(){

clearTimeout(idleTimer);

stopIdleMode();

idleTimer =
setTimeout(startIdleMode,30000);

}

[
"mousemove",
"mousedown",
"keypress",
"touchstart",
"scroll"
].forEach(event => {

document.addEventListener(
event,
resetIdleTimer
);

});

resetIdleTimer();
